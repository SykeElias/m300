=== Apache2

``sudo apt install apache2
``

**Apache aktivieren:
**
image::27EnableApache.png[]

image::28ApacheResources.png[]

**VirtualHost konfigurieren:
**

sudo nano /etc/apache2/sites-available/myweb.conf

    <VirtualHost *:80>
        ServerName myweb.local
        DocumentRoot /var/www/html/myweb
        ErrorLog ${APACHE_LOG_DIR}/error.log
        CustomLog ${APACHE_LOG_DIR}/access.log combined
    </VirtualHost>


    sudo a2ensite myweb
    sudo a2dissite 000-default    # Standard-Site deaktivieren
    sudo systemctl reload apache2

*Testen*

    wget -qO- http://192.168.220.13

*Debugging*

    sudo systemctl status apache2      # Status prüfen
    sudo apache2ctl -S                 # VirtualHost-Konfiguration anzeigen
    sudo ss -tlpn | grep :80          # Port 80 prüfen
    sudo journalctl -xeu apache2      # Logs bei Fehler

== Webmin

image::33BIND.png[]

    sudo docker exec dns-webmin-bind-1 bash -c 'cat > /etc/bind/named.conf.local << EOF
    // lokale zonen
    //
    zone "lan.m300.smartlearn.ch" {
        type master;
        file "/etc/bind/db.lan.m300.smartlearn.ch";
    };
    zone "220.168.192.in-addr.arpa" {
        type master;
        file "/etc/bind/db.192.168.220";
    };
    zone "210.168.192.in-addr.arpa" {
        type master;
        file "/etc/bind/db.192.168.210";
    };
    EOF'


Zonen Records anpassen:

    sudo docker exec dns-webmin-bind-1 bash -c 'cat > /etc/bind/db.192.168.220 << EOF
    \$TTL 604800
    @   IN  SOA vmls3.lan.m300.smartlearn.ch. admin.lan.m300.smartlearn.ch. (
                2026042601 ; Serial
                604800     ; Refresh
                86400      ; Retry
                2419200    ; Expire
                604800 )   ; Negative Cache TTL
    @   IN  NS  vmls3.lan.m300.smartlearn.ch.
    13  IN  PTR vmls3.lan.m300.smartlearn.ch.
    EOF'


    sudo docker exec dns-webmin-bind-1 bash -c 'cat > /etc/bind/db.192.168.210 << EOF
    \$TTL 86400
    @   IN SOA ns.m300.smartlearn.ch. root.m300.smartlearn.ch. (
                            2026042601
                            14400
                            1800
                            1209600
                            3600 )
    @   172800  IN NS ns.m300.smartlearn.ch.
    1   IN PTR vmlf1.lan.m300.smartlearn.ch.
    31  IN PTR vmlp1.lan.m300.smartlearn.ch.
    EOF'


    sudo docker exec dns-webmin-bind-1 bash -c 'cat > /etc/bind/db.lan.m300.smartlearn.ch << EOF
    \$TTL 3600
    lan.m300.smartlearn.ch. IN SOA ns.m300.smartlearn.ch. root.m300.smartlearn.ch. (
                            2026042601
                            3600
                            600
                            1209600
                            3600 )
    lan.m300.smartlearn.ch. IN NS ns.m300.smartlearn.ch.
    vmls3   IN A 192.168.220.13
    vmlp1   IN A 192.168.210.31
    test    IN A 3.4.5.6
    EOF'
